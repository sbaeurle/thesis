\usepackage[pdftitle={\vartitle},pdfauthor={\varname},pdfsubject={\vartype},pdfdisplaydoctitle=true]{hyperref}	% package to enable hyperlinks in the document

% predefined colors
\newcommand{\varlinkcolor}{blue}	% link color in the document

% resource paths
\graphicspath{{./images/}}

% hyperref setup
\hypersetup{
	colorlinks=true,		% colored links
	linkcolor=\varlinkcolor,% link color
	citecolor=\varlinkcolor,% citation color
	filecolor=\varlinkcolor,% file link color
	menucolor=\varlinkcolor,% menu link color
	urlcolor=\varlinkcolor,	% url color
	linktocpage=true, 		% enables clickable packenumbers in the toc
	bookmarksnumbered=true 	% adds sections to the pdf structure
}

% biblatex config
\usepackage[
	backend=biber,		
	bibwarn=true,
	bibencoding=utf8,
	sortlocale=de_DE,
	sorting=none,
	style=numeric-comp
]{biblatex}

\addbibresource{literature.bib}

% code listing configuration

% color definitions
\definecolor{lst-background}{RGB}{252,252,252}
\definecolor{lst-keyword}{RGB}{34,34,153}
\definecolor{lst-comment}{RGB}{34,139,34}
\definecolor{lst-string}{RGB}{160,32,240}

% Standard
\lstset{
		language=Java,			% standard language
		numbers=left,			% line number position
		stepnumber=1,			% line number steps
		numbersep=5pt,			% distance to code
		numberstyle=\tiny,		% style of the number
		breaklines=true,		% automatic linebreak
		breakautoindent=true,	% indentation after linebreak
		postbreak=\space,		% linebreak after space
		tabsize=2,				% size of the tabulator
		basicstyle=\ttfamily\footnotesize, % styling of the code
		showspaces=false,		% do not show spaces
		showstringspaces=false,	% in strings too
		extendedchars=true,		% latin1 charset
		captionpos=b,			% caption at the bottom
		backgroundcolor=\color{lst-background}, % background color.
		xleftmargin=0pt,		% left margin
		xrightmargin=0pt,		% right margin
		frame=single,			% frame setting
		frameround=ffff,		% frame corner style (t round / f sharp)
		rulecolor=\color{darkgray},			% frame color
		fillcolor=\color{lst-background},	% filling color
		keywordstyle=\color{lst-keyword},	% keyword color
		commentstyle=\color{lst-comment},	% comment color
		stringstyle=\color{lst-string}		% string color
}
